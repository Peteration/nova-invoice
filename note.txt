-- User table
create table users (
  id uuid primary key default gen_random_uuid(),
  email text unique,
  created_at timestamp default now()
);

-- Invoices
create table invoices (
  id uuid primary key default gen_random_uuid(),
  user_id uuid references users(id),
  client_name text,
  amount text,
  upgraded boolean default false,
  created_at timestamp default now()
);
